VLIB 相关源码目录： ./src/{vlib, vlibapi, vlibmemory}。

VLIB库提供了了向量处理支持，包括： graph-node调度、可靠组播支持、轻量级的多任务协程、CLI、.DLL插件、物理内存以及linux epoll。

1. 初始化函数

vlib应用通过注册各种 [initialization] 事件，将数据结构以及带有 __attribute__((constructor)) 属性的函数放置到程序映像中。在合适的时间，vlib框架遍历 构造函数，生成单链结构链表，并根据指定的约束对其进行排序，并调用指示的函数。 vlib 应用通过该机制实现许多功能：创建 graph nodes，添加CLI功能，启动多任务协程等等。

vlib应用包含了许多 VLIB_INIT_FUNCTION (my_init_function) 宏。

每个 init / configure / 等等 函数的返回值类型是 clib_error_t * 。函数执行成功返回0，其他返回值会导致框架报错并退出。

vlib应用必须链接 vppinfra库，并且通常也会链接到其他库（VNET）。

初始化构造函数、约束规范： 

static clib_error_t *my_init_function (vlib_main_t *vm)
{
   /* ... initialize things ... */

   return 0; // or return clib_error_return (0, "BROKEN!");
}
VLIB_INIT_FUNCTION(my_init_function);

以上示例中，my_init_function会在特定的点被执行，但是没有约定顺序。

添加顺序约束： 

VLIB_INIT_FUNCTION(my_init_function) =
{
   .runs_before = VLIB_INITS("we_run_before_function_1",
                             "we_run_before_function_2"),
   .runs_after = VLIB_INITS("we_run_after_function_1",
                            "we_run_after_function_2),
 };

批量指定从 a -> b -> c -> d 的顺序： 

VLIB_INIT_FUNCTION(my_init_function) =
{
   .init_order = VLIB_INITS("a", "b", "c", "d"),
};

2. Vector 数据结构

vpp/vlib 中，我们使用 vlib_frame_t 类型来描述 vector 实例： 

typedef struct vlib_frame_t
{
  /* Frame flags. */
  u16 flags;

  /* Number of scalar bytes in arguments. */
  u8 scalar_size;

  /* Number of bytes per vector argument. */
  u8 vector_size;

  /* Number of vector elements currently in frame. */
  u16 n_vectors;

  /* Scalar and vector arguments to next node. */
  u8 arguments[0];
} vlib_frame_t;

可以构造任意类型的 vectors ，带有一些相关的 scalar 数据也可以使用该结构构建。vpp应用中，vectors通常使用 4 字节来描述vector 元素大小，0字节描述单个frame的scalar数据。

Frame 结构对齐 CLIB_CACHE_LINE_BYTES 边界。

4.1. command注册

#define VLIB_CLI_COMMAND(x, ...)                                               \
        __VA_ARGS__ vlib_cli_command_t x;                                      \
        static void __vlib_cli_command_registration_##x(void)                  \
                __attribute__((__constructor__));                              \
        static void __vlib_cli_command_registration_##x(void)                  \
        {                                                                      \
                vlib_main_t *vm = vlib_get_main();                             \
                vlib_cli_main_t *cm = &vm->cli_main;                           \
                x.next_cli_command = cm->cli_command_registrations;            \
                cm->cli_command_registrations = &x;                            \
        }                                                                      \
        static void __vlib_cli_command_unregistration_##x(void)                \
                __attribute__((__destructor__));                               \
        static void __vlib_cli_command_unregistration_##x(void)                \
        {                                                                      \
                vlib_main_t *vm = vlib_get_main();                             \
                vlib_cli_main_t *cm = &vm->cli_main;                           \
                VLIB_REMOVE_FROM_LINKED_LIST(cm->cli_command_registrations,    \
                                             &x, next_cli_command);            \
        }                                                                      \
        __VA_ARGS__ vlib_cli_command_t x

4.2. 宏定义

利用gcc attribute__((__constructor)) 属性进行构造初始化 

VLIB_INIT_FUNCTION(dpdk_main_init) = {
        .runs_after = VLIB_INITS("dpdk_init"),
};

VLIB_DECLARE_INIT_FUNCTION(x, tag)

vlib_init_function_t *_VLIB_INIT_FUNCTION_SYMBOL(x, tag) = dpdk_main_init;
static void __vlib_add_init_function_dpdk_main_init(void)
        __attribute__((__constructor__));
static _vlib_init_function_list_elt_t _vlib_init_function_init_dpdk_main_init;
static void __vlib_add_init_function_dpdk_main_init(void)
{
        vlib_main_t *vm = vlib_get_main();
        _vlib_init_function_init_dpdk_main_init.next_init_function =
                vm->init_function_registrations;
        vm->init_function_registrations =
                &_vlib_init_function_init_dpdk_main_init;
        _vlib_init_function_init_dpdk_main_init.f = &dpdk_main_init;
        _vlib_init_function_init_dpdk_main_init.name = "dpdk_main_init";
}
static void __vlib_rm_init_function_dpdk_main_init(void)
        __attribute__((__destructor__));
static void __vlib_rm_init_function_dpdk_main_init(void)
{
        vlib_main_t *vm = vlib_get_main();
        _vlib_init_function_list_elt_t *this, *prev;
        this = vm->init_function_registrations;
        if (this == 0)
                return;
        if (this->f == dpdk_main_init) {
                vm->init_function_registrations =
                        this->next_init_function;
                return;
        }
        prev = this;
        this = this->next_init_function;
        while (this) {
                if (this->f == dpdk_main_init) {
                        prev->next_init_function =
                                this->next_init_function;
                        return;
                }
                prev = this;
                this = this->next_init_function;
        }
}
static _vlib_init_function_list_elt_t _vlib_init_function_init_dpdk_main_init  = {
        .runs_after = VLIB_INITS("dpdk_init"),
};
