1. Socket函数

#include <sys/types.h>          /* See NOTES */
#include <sys/socket.h>

int socket(int domain, int type, int protocol);

/**
 * enum sock_type - Socket types
 * @SOCK_STREAM: stream (connection) socket
 * @SOCK_DGRAM: datagram (conn.less) socket
 * @SOCK_RAW: raw socket
 * @SOCK_RDM: reliably-delivered message
 * @SOCK_SEQPACKET: sequential packet socket
 * @SOCK_DCCP: Datagram Congestion Control Protocol socket
 * @SOCK_PACKET: linux specific way of getting packets at the dev level.
 *                For writing rarp and other similar things on the user level.
 *
 * When adding some new socket type please
 * grep ARCH_HAS_SOCKET_TYPE include/asm-* /socket.h, at least MIPS
 * overrides this enum for binary compat reasons.
 */
enum sock_type {
        SOCK_STREAM     = 1,
        SOCK_DGRAM      = 2,
        SOCK_RAW        = 3,
        SOCK_RDM        = 4,
        SOCK_SEQPACKET  = 5,
        SOCK_DCCP       = 6,
        SOCK_PACKET     = 10,
};

2. sendmsg和recvmsg的高级用法

/* glibc库中的定义 */

/* 用来存储辅助数据对象的结构 */
struct cmsghdr
{
        socklen_t cmsg_len;             /* cmsg_data的数据长度加cmsghdr结构的长度 */
        int cmsg_level;         /* 原始协议 */
        int cmsg_type;          /* 指定协议类型  */
};

/* Structure for scatter/gather I/O.  */
struct iovec
{
        void *iov_base; /* 数据的起始地址  */
        size_t iov_len; /* 数据长度  */
};

struct msghdr
{
        void            *msg_name;      /* 指向一个套接字的地址结构 */
        int                 msg_namelen;        /* msg_name指向的地址结构大小 */

        struct iovec *msg_iov;  /* 指定输入缓冲区或者输出缓冲区的数组 */
        int msg_iovlen;         /* 输入或输出缓冲区的数组长度  */

        void *msg_control;              /* 辅助数据的地址 */
        socklen_t msg_controllen;       /* 辅助数据的长度  */

        int msg_flags;          /* 接受消息的flag  */
};

/* Bits in the FLAGS argument to `send', `recv', et al.  */
enum
{
        MSG_OOB         = 0x01, /* Process out-of-band data.  */
        MSG_PEEK                = 0x02, /* Peek at incoming messages.  */
        MSG_DONTROUTE   = 0x04, /* Don't use local routing.  */
        MSG_EOR         = 0x08, /* Data completes record.  */
        MSG_TRUNC               = 0x10, /* Data discarded before delivery.  */
        MSG_CTRUNC              = 0x20, /* Control data lost before delivery.  */
        MSG_WAITALL             = 0x40, /* Wait for full request or error.  */
        MSG_DONTWAIT    = 0x80, /* This message should be nonblocking.  */
        MSG_NOSIGNAL    = 0x0400        /* Do not generate SIGPIPE on EPIPE.  */
};


#include <sys/types.h>
#include <sys/socket.h>
ssize_t sendmsg(int sockfd, const struct msghdr *msg, int flags);
ssize_t recvmsg(int sockfd, struct msghdr *msg, int flags);

2.1. 传递文件描述符

通过UNIX域套接字，可以将包含文件描述符的辅助数据从一个进程传递到另一个进程上。通过这种方式可以传递任意类型的文件描述符（例如open或者pipe等返回的文件描述符）。

例如在一个tcp服务器监听了端口等待客户端连接，当服务器接受连接后，可以将连接发送给子进程处理连接上的事件。两个进程中的fd一般不相同（fd代表本进程中文件描述符列表中的位置），但实际上底层是对同一个打开文件描述符的引用。
